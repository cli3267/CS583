2. Implementation of Logistic Regression¶

2.1. Worker Node Class¶

In [5]:
class Worker:
    def __init__(self, x, y):
        self.x = x # s by d local feature matrix
        self.y = y # s by 1 local label matrix
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features
        self.w = numpy.zeros((d,1)) # d by 1 model parameter vector

    
    # Set the model parameters to the latest
    def set_param(self, w):
        self.w = w
        
    # Compute the local loss
    def loss(self):
        yx = numpy.multiply(self.y, self.x)
        ywx = numpy.dot(yx, self.w) 
        vec1 = numpy.exp(-ywx) 
        vec2 = numpy.log(1 + vec1) 
        return numpy.sum(vec2) 
    
    # Compute the local gradient
    def gradient(self):
        yx = numpy.multiply(self.y, self.x) 
        ywx = numpy.dot(yx, self.w) 
        vec1 = numpy.exp(ywx) 
        vec2 = numpy.divide(yx, 1 + vec1) 
        g = -numpy.sum(vec2, axis=0).reshape(self.d, 1) 
        return g
    
    # below, I have used my homework 2 as a resource
    # Gradient descent for solving logistic regression
    def gradientDescent(self, q, stepsize):
        wk = self.w # used to store the inital w for computation at the end
        for t in range(q):
            g = self.gradient()
            self.w -= stepsize * g
        return wk - self.w # Computing the ascending direction by the kth worker
    
    # Compute the local SGD
    def stochasticObjectiveGradient(self, xi, yi, lam):
        yx = numpy.multiply(yi, xi) # 1 by d matrix
        yxw = float(numpy.dot(yx, self.w)) #scalar
        g_loss = -yx.T / (1 + numpy.exp(yxw)) # d-by-1 matrix
        g = g_loss + lam * self.w
        return g

    # SGD for solving logistic regression
    def stochasticGradientDescent(self, q, lam, stepsize):
        for t in range(q):
            index = numpy.random.permutation(self.s)
            x_rand = self.x[index, :]
            y_rand = self.y[index, :] 
                        
            for i in range(self.s):
                xi = x_rand[i, :]
                yi = float(y_rand[i, :])
                g = self.stochasticObjectiveGradient(xi, yi, lam)
                self.w -= stepsize * g
            stepsize *= 0.7
        return self.w

2.2 Server Class¶

In [6]:
class Server:
    def __init__(self, m, n, d):
        self.m = m # number of worker nodes
        self.n = n # number of training samples
        self.d = d # number of features
        self.w = numpy.zeros((d,1)) # d by 1 model parameter vector
        self.g = numpy.zeros((d,1)) # d by 1 gradient
        self.v = numpy.zeros((d,1)) # d by 1 momentum
        self.loss = 0 # loss function value
        self.obj = 0 # objective function value
    
    def broadcast(self):
        return self.w
    
    # Sum the gradients and loss functions evaluated by the workers
    def aggregate(self, grads, losses):
        self.g = numpy.zeros((self.d, 1))
        self.loss = 0
        for k in range(self.m):
            self.g += grads[k]
            self.loss += losses[k]
        self.g = self.g / self.m
        
    # Computer the gradident
    def gradient(self, lam):
        self.g = self.g / self.n + lam * self.w
    
    # Computer the objective function
    def objective(self, lam):
        reg = lam / 2 * numpy.sum(self.w * self.w)
        self.obj = self.loss / self.n + reg
        return self.obj

    # Update the model paramteres using agd
    def agd(self, apha, beta):
        self.v *= beta
        self.v += self.g
        self.w -= alpha * self.v
      
    # Update the model parameters
    def updateGD(self, alpha):
        self.w -= alpha * self.g

import math

m = 4 # number of worker nodes
# Create a server and m worker nodes
def create_server_workers(x, y):
    n,d = x.shape
    s = math.floor(n / m)
    server = Server(m, n, d)
    workers = []
    
    for i in range(m):
        index = list(range(i * s, (i +  1) * s))
        worker = Worker(x[index, :], y[index, :])
        workers.append(worker)
        
    return server, workers

2.4 Run Federated Averaging¶

In [8]:
lam = 1E-6 # regularization parameter
alpha = 1E-3 # learning rate
max_epoch = 50 # number of epoch

def runFedAveraging(q, formula='GD'):
    objvals = numpy.zeros(max_epoch)
    for t in range(max_epoch):
        # Broadcast
        w = server.broadcast()
        for i in range(m):
            workers[i].set_param(w)

        # Workers' local computations
        grads = []
        losses = []

        for i in range(m):
            # Determines what formula to run
            if(formula == 'GD'):
                g = workers[i].gradientDescent(q, alpha)
            else:
                g = workers[i].stochasticGradientDescent(q, lam, alpha)
            l = workers[i].loss()
            grads.append(g)
            losses.append(l)

        # Aggregates the workers' outputs
        server.aggregate(grads, losses)

        # Server updates the model parameters
        server.gradient(lam)
        obj = server.objective(lam)

        print('Objective function value = ' + str(obj))
        objvals[t] = obj
        server.updateGD(alpha)

    return objvals
